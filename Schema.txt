SCHEMA FILE : 

1. CREATING FILM TABLE : 
CREATE TABLE film (
id INT PRIMARY KEY,
title VARCHAR(255),
release_year INT,
genre VARCHAR(255),
score INT CHECK (score >= 0 AND score <= 10)
);

2. CREATING DIRECTOR TABLE :
CREATE TABLE director (
id INT PRIMARY KEY,
name VARCHAR(255) UNIQUE NOT NULL,
country VARCHAR(255)
);

3. ADDING DIRECTOR_ID COLUMN TO FILM TABLE :
ALTER TABLE film
ADD COLUMN director_id INT;

4. ADDING FOREIGN KEY CONSTRAINT :
ALTER TABLE film
ADD CONSTRAINT fk_director
FOREIGN KEY (director_id) REFERENCES directors(id);

5. CREATING STAR TABLE : 
CREATE TABLE star (
id INT PRIMARY KEY,
name VARCHAR(255) UNIQUE NOT NULL,
DoB INTEGER
);

6. ADDING STAR_ID COLUMN TO FILM TABLE :
ALTER TABLE film
ADD COLUMN star_id INTEGER;


7 .ADDING FOREIGN KEY CONSTRAINT FOR STAR :
ALTER TABLE film
ADD CONSTRAINT fk_star
FOREIGN KEY (star_id) REFERENCES star(id);


8. CREATING WRITER TABLE :
CREATE TABLE writer (
id INT PRIMARY KEY,
name VARCHAR(255) UNIQUE NOT NULL,
email VARCHAR(255) UNIQUE NOT NULL
);


9. ADDING WRITER_ID COLUMN TO FILM TABLE :
ALTER TABLE film ADD COLUMN writer_id INT;

10. ADDING FOREIGN KEY CONSTRAINT FOR WRITER :

ALTER TABLE film 
ADD CONSTRAINT fk_writers
FOREIGN KEY(writer_id) REFERENCES writers(id);
